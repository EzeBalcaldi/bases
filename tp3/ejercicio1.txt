ejercicio 1.2
select nombre, apellido, sueldo, e_mail
from empleado
where sueldo > 1000 and e_mail LIKE '%@gmail.com'
order by sueldo
limit 100

ejercicio 1.3
select id_tarea
from empleado
group by id_tarea
order by id_tarea ASC

ejercicio 1.4
select nombre, apellido, telefono, id_tarea
from empleado
where id_tarea = '7231'
order by apellido, nombre
